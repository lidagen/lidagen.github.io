git commit <fileName> -m "commit content" жҸҗдәӨж–Үд»¶
git push 'https://xxx.git' master жҺЁйҖҒmaster

git log жҹҘзңӢжҸҗдәӨи®°еҪ•
git reset --hard HEAD^ еӣһйҖҖдёҠдёҖдёӘзүҲжң¬
git reset --hard HEAD~100 еӣһйҖҖдёҠ100дёӘзүҲжң¬
git reset --hard <зүҲжң¬еҸ·> еӣһйҖҖеӣһеҲ°жҹҗдёӘжңӘжқҘзүҲжң¬
git reflog жҹҘзңӢе‘Ҫд»ӨеҺҶеҸІ
git checkout В --<fileName> дёўејғе·ҘдҪңеҺ»дҝ®ж”№
git rm <fileName> еҲ йҷӨж–Үд»¶(иҝҳйңҖиҰҒcommit)

git reset --soft HEAD^ ж’Өй”ҖдёҠдёҖж¬Ўcommit
git reset --soft HEAD~2 ж’Өй”ҖдёҠ2ж¬Ўcommit

е·Із»Ҹpushзҡ„д»Јз Ғ
reset current branch
git push --force

resetеҮ дёӘеҸӮж•°пјҡ

--mixed 
ж„ҸжҖқжҳҜпјҡдёҚеҲ йҷӨе·ҘдҪңз©әй—ҙж”№еҠЁд»Јз ҒпјҢж’Өй”ҖcommitпјҢе№¶дё”ж’Өй”Җgit add . ж“ҚдҪң
иҝҷдёӘдёәй»ҳи®ӨеҸӮж•°,git reset --mixed HEAD^ е’Ң git reset HEAD^ ж•ҲжһңжҳҜдёҖж ·зҡ„гҖӮ

--soft
дёҚеҲ йҷӨе·ҘдҪңз©әй—ҙж”№еҠЁд»Јз ҒпјҢж’Өй”ҖcommitпјҢдёҚж’Өй”Җgit add . 

--hard
еҲ йҷӨе·ҘдҪңз©әй—ҙж”№еҠЁд»Јз ҒпјҢж’Өй”ҖcommitпјҢж’Өй”Җgit add . 
жіЁж„Ҹе®ҢжҲҗиҝҷдёӘж“ҚдҪңеҗҺпјҢе°ұжҒўеӨҚеҲ°дәҶдёҠдёҖж¬Ўзҡ„commitзҠ¶жҖҒ

йЎәдҫҝиҜҙдёҖдёӢпјҢеҰӮжһңcommitжіЁйҮҠеҶҷй”ҷдәҶпјҢеҸӘжҳҜжғіж”№дёҖдёӢжіЁйҮҠпјҢеҸӘйңҖиҰҒпјҡ
git commit --amend

# еҲӣе»әеҲҶж”Ҝе’ҢеҗҲе№¶еҲҶж”Ҝ

git checkout -b dev еҲӣе»әdevеҲҶж”Ҝе№¶еҲҮжҚўеҲ°devеҲҶж”Ҝ -b иЎЁзӨәеҲӣе»әе№¶еҲҮжҚў,зӯүдәҺ git branch dev + git checkout dev дёӨжқЎе‘Ҫд»Ө
git branch жҹҘзңӢжң¬ең°еҲҶж”Ҝ
git branch -r жҹҘзңӢиҝңзЁӢеҲҶж”Ҝ
git checkout <еҲҶж”ҜеҗҚ> еҲҮжҚўеҲҶж”Ҝ
git merge <йңҖиҰҒеҗҲе№¶зҡ„еҲҶж”ҜеҗҚ> еҗҲе№¶еҲҶж”ҜеҲ°еҪ“еүҚеҲҶж”Ҝ Fast ForwordжЁЎејҸ

# и§ЈеҶіеҶІзӘҒжҖқи·Ҝ

1.git merge,е…ҲmergeдёӨдёӘзүҲжң¬
2.жүӢеҠЁи§ЈеҶіеҶІзӘҒ(git з”Ё<<<<<<< , =======, <<<<<<<ж Үи®°дёҚеҗҢеҲҶж”Ҝзҡ„еҶ…е®№,жүӢеҠЁйҖүжӢ©и§ЈеҶіиЎҢеҶІзӘҒзҡ„ең°ж–№)
3.з„¶еҗҺaddгҖҒcommitжҸҗдәӨ
4.git log --graph --pretty=oneline --abbrev-commitеҸҜд»ҘзңӢеҲҶж”ҜеҗҲе№¶жғ…еҶө

# еҲҶж”Ҝз®ЎзҗҶзӯ–з•Ҙno-ffжЁЎејҸ

1.йҖҡеёёеҗҲе№¶еҲҶж”Ҝ,GITз”Ёзҡ„жҳҜfast forwordжЁЎејҸ,иҝҷз§ҚжЁЎејҸдёӢ,еҲ йҷӨеҲҶж”ҜеҗҺ,дјҡдёўејғеҲҶж”ҜдҝЎжҒҜгҖӮ
2.git merge --no-f -m "commit content" <еҲҶж”ҜеҗҚ> В  В no ffеҗҲе№¶
3.еҗҲе№¶еҲҶж”Ҝж—¶пјҢеҠ дёҠ--no-ffеҸӮж•°е°ұеҸҜд»Ҙз”Ёжҷ®йҖҡжЁЎејҸеҗҲе№¶пјҢеҗҲе№¶еҗҺзҡ„еҺҶеҸІжңүеҲҶж”ҜпјҢиғҪзңӢеҮәжқҘжӣҫз»ҸеҒҡиҝҮеҗҲе№¶пјҢ
  иҖҢfast forwardеҗҲе№¶е°ұзңӢдёҚеҮәжқҘжӣҫз»ҸеҒҡиҝҮеҗҲе№¶гҖӮ

# GITејҖеҸ‘й”ҷеҲҶж”ҜеӨ„зҗҶ

1.д»Јз Ғж”№дәҶиҝҳжІЎжҸҗдәӨ

git add . (жҠҠжүҖжңүж”№еҠЁжҡӮеӯҳ) git stash (жҠҠжҡӮеӯҳзҡ„ж–Үд»¶жҸҗдәӨеҲ°gitзҡ„жҡӮеӯҳж Ҳ) git checkout жң¬иҜҘжҸҗдәӨд»Јз Ғзҡ„еҲҶж”Ҝ git stash pop (е°ҶжҡӮеӯҳж Ҳдёӯзҡ„д»Јз Ғж”ҫеҮәжқҘ)

2.д»Јз Ғж”№дәҶиҝҳжҸҗдәӨдәҶ

git checkout дёҚиҜҘжҸҗдәӨд»Јз ҒжҸҗдәӨдәҶд»Јз Ғзҡ„еҲҶж”Ҝ git reset HEAD~1 пјҲжңҖиҝ‘дёҖж¬ЎжҸҗдәӨж”ҫеӣһжҡӮеӯҳеҢә, е№¶еҸ–ж¶ҲжӯӨж¬ЎжҸҗдәӨпјү git stash (жҠҠжҡӮеӯҳзҡ„ж–Үд»¶жҸҗдәӨеҲ°gitзҡ„жҡӮеӯҳж Ҳ) git checkout иҜҘжҸҗдәӨд»Јз Ғзҡ„еҲҶж”Ҝ git stash pop зӯүдҪ жҠҠд»Јз ҒжҸҗдәӨеҲ°дәҶжӯЈзЎ®зҡ„еҲҶж”ҜеҗҺпјҢеҶҚж¬ЎеҲҮеҲ°еҲҡеҲҡй”ҷзҡ„еҲҶж”Ҝ git push origin й”ҷиҜҜзҡ„еҲҶж”Ҝ -f (жҠҠдёҚиҜҘдёҠеҺ»зҡ„ж–Үд»¶еӣһйҖҖжҺү)

# bugеҲҶж”Ҝ

1.еңәжҷҜпјҡеҪ“еүҚdevеҲҶж”Ҝе·ҘдҪңжңӘе®ҢжҲҗ,дёҚжғіиҰҒжҸҗдәӨд»Јз Ғ,дҪҶжҳҜжӯӨж—¶жңүдёӘbugеҲҶж”ҜйңҖиҰҒдҪ зҙ§жҖҘеӨ„зҗҶ
2.git stash жҡӮеӯҳdevзӣ®еүҚдҝ®ж”№жңӘжҸҗдәӨзҡ„д»Јз Ғ,иҝҷж—¶еҸҜд»ҘеҲҮжҚўbugеҲҶж”ҜиҝӣиЎҢbugдҝ®еӨҚ
3.bugдҝ®еӨҚе®ҢжҸҗдәӨеҗҲе№¶,еҲҮеӣһdevеҲҶж”ҜпјҢgit stash listе‘Ҫд»ӨеҸҜжҹҘзңӢжҡӮеӯҳзҡ„е·ҘдҪңзҺ°еңә
4.жҒўеӨҚпјҡ1.git stash apply жҒўеӨҚ,жҒўеӨҚеҗҺstashеҶ…е®№е№¶дёҚеҲ йҷӨпјҢдҪ йңҖиҰҒgit stash dropжқҘеҲ йҷӨ В 2.git stash pop 
  жҒўеӨҚзҡ„еҗҢж—¶еҲ йҷӨжҡӮеӯҳ
5.еҰӮжһңдҪ жңүеӨҡж¬ЎжҡӮеӯҳ,жҒўеӨҚзҡ„ж—¶еҖҷgit stash listжҹҘзңӢ,жҢҮе®ҡжҒўеӨҚзҡ„stash,git stash apply <жҡӮеӯҳеҸ·>еҰӮ:stash@{0}

# featureеҲҶж”Ҝ

1.еңәжҷҜпјҡж·»еҠ дёҖдёӘе®һйӘҢжҖ§ж–°еҠҹиғҪ,дҪ дёҚеёҢжңӣжҠҠдё»еҲҶж”Ҝжҗһд№ұдәҶ,иҝҷж—¶еҖҷжңҖеҘҪж–°е»әдёҖдёӘfeatureеҲҶж”Ҝ,еңЁдёҠйқўејҖеҸ‘гҖӮ
2.ејҖеҸ‘е®ҢжңӘеҗҲе№¶зҡ„жғ…еҶөдёӢ,ж–°еҠҹиғҪеҸ–ж¶Ҳ,йңҖиҰҒеҲ йҷӨеҲҶж”Ҝ,иҝҷж—¶git branch -d <еҲҶж”ҜеҗҚ>еҲ йҷӨеҲҶж”Ҝдјҡеӣ дёәжңӘеҗҲе№¶еӨұиҙҘ,жӣҙж”№е‘Ҫд»Ө 
  git branch -D <еҲҶж”ҜеҗҚ> ејәиЎҢеҲ йҷӨ

# еӨҡдәәеҚҸдҪң

1.иҝңзЁӢеә“cloneж—¶,е®һйҷ…GITжҳҜжҠҠжң¬ең°masterеҲҶж”ҜдёҺиҝңзЁӢmasterеҜ№еә”иө·жқҘдәҶ,е№¶дё”иҝңзЁӢеә“й»ҳи®ӨеҗҚз§°жҳҜorigin
2.git remote -v жҹҘзңӢиҝңзЁӢеә“дҝЎжҒҜ,е№¶жҹҘзңӢжқғйҷҗ
3.жҺЁйҖҒеҲҶж”Ҝ,е°ұжҳҜжҠҠиҜҘеҲҶж”ҜжүҖжңүжң¬ең°жҸҗдәӨжҺЁйҖҒеҲ°иҝңзЁӢеә“ git push origin master
4.жҠ“еҸ–еҲҶж”Ҝ,й»ҳи®Өзҡ„ git clone е…ӢйҡҶзҡ„жҳҜmasterеҲҶж”Ҝ,дҪ йңҖиҰҒеңЁdevејҖеҸ‘,git checkout -b dev origin/dev,
  еҲӣе»әиҝңзЁӢoriginзҡ„devеҲҶж”ҜеҲ°жң¬ең°гҖӮ

# еӨҡдәәеҚҸдҪңе·ҘдҪңжЁЎејҸ

1.йҰ–е…ҲпјҢеҸҜд»ҘиҜ•еӣҫз”Ёgit push origin <branch-name>жҺЁйҖҒиҮӘе·ұзҡ„дҝ®ж”№пјӣ
2.еҰӮжһңжҺЁйҖҒеӨұиҙҘпјҢеҲҷеӣ дёәиҝңзЁӢеҲҶж”ҜжҜ”дҪ зҡ„жң¬ең°жӣҙж–°пјҢйңҖиҰҒе…Ҳз”Ёgit pullиҜ•еӣҫеҗҲе№¶пјӣ
3.еҰӮжһңеҗҲе№¶жңүеҶІзӘҒпјҢеҲҷи§ЈеҶіеҶІзӘҒпјҢе№¶еңЁжң¬ең°жҸҗдәӨпјӣ
4.жІЎжңүеҶІзӘҒжҲ–иҖ…и§ЈеҶіжҺүеҶІзӘҒеҗҺпјҢеҶҚз”Ёgit push origin <branch-name>жҺЁйҖҒе°ұиғҪжҲҗеҠҹпјҒ
  еҰӮжһңgit pullжҸҗзӨәno tracking informationпјҢеҲҷиҜҙжҳҺжң¬ең°еҲҶж”Ҝе’ҢиҝңзЁӢеҲҶж”Ҝзҡ„й“ҫжҺҘе…ізі»жІЎжңүеҲӣе»әпјҢ
  з”Ёе‘Ҫд»Өgit branch --set-upstream-to <branch-name> origin/<branch-name>гҖӮ

# жҗӯе»әGITжңҚеҠЎеҷЁ

1.жҗӯе»әGitжңҚеҠЎеҷЁйңҖиҰҒеҮҶеӨҮдёҖеҸ°иҝҗиЎҢLinuxзҡ„жңәеҷЁпјҢејәзғҲжҺЁиҚҗз”ЁUbuntuжҲ–DebianпјҢиҝҷж ·пјҢйҖҡиҝҮеҮ жқЎз®ҖеҚ•зҡ„aptе‘Ҫд»Өе°ұеҸҜд»Ҙе®ҢжҲҗе®үиЈ…гҖӮ
2.еҒҮи®ҫдҪ е·Із»Ҹжңүsudoжқғйҷҗзҡ„з”ЁжҲ·иҙҰеҸ·пјҢдёӢйқўпјҢжӯЈејҸејҖе§Ӣе®үиЈ…гҖӮ
В В  В з¬¬дёҖжӯҘпјҢе®үиЈ…gitпјҡsudo apt-get install git
В В  В з¬¬дәҢжӯҘпјҢеҲӣе»әдёҖдёӘgitз”ЁжҲ·пјҢз”ЁжқҘиҝҗиЎҢgitжңҚеҠЎпјҡ sudo adduser git
В В  В з¬¬дёүжӯҘпјҢеҲӣе»әиҜҒд№Ұзҷ»еҪ•пјҡж”¶йӣҶжүҖжңүйңҖиҰҒзҷ»еҪ•зҡ„з”ЁжҲ·зҡ„е…¬й’ҘпјҢе°ұжҳҜд»–д»¬иҮӘе·ұзҡ„id_rsa.pubж–Үд»¶пјҢжҠҠжүҖжңүе…¬й’ҘеҜје…ҘеҲ°/home/git/.ssh/authorized_keysж–Үд»¶йҮҢпјҢдёҖиЎҢдёҖдёӘгҖӮ
В В  В з¬¬еӣӣжӯҘпјҢеҲқе§ӢеҢ–Gitд»“еә“пјҡе…ҲйҖүе®ҡдёҖдёӘзӣ®еҪ•дҪңдёәGitд»“еә“пјҢеҒҮе®ҡжҳҜ/srv/sample.gitпјҢеңЁ/srvзӣ®еҪ•дёӢиҫ“е…Ҙе‘Ҫд»Өпјҡ sudo git init --bare sample.git
В В  В Gitе°ұдјҡеҲӣе»әдёҖдёӘиЈёд»“еә“пјҢиЈёд»“еә“жІЎжңүе·ҘдҪңеҢәпјҢеӣ дёәжңҚеҠЎеҷЁдёҠзҡ„Gitд»“еә“зәҜзІ№жҳҜдёәдәҶе…ұдә«пјҢжүҖд»ҘдёҚи®©з”ЁжҲ·зӣҙжҺҘзҷ»еҪ•еҲ°жңҚеҠЎеҷЁдёҠеҺ»ж”№е·ҘдҪңеҢәпјҢе№¶дё”жңҚеҠЎеҷЁдёҠзҡ„Gitд»“еә“йҖҡеёёйғҪд»Ҙ.gitз»“е°ҫгҖӮз„¶еҗҺпјҢжҠҠownerж”№дёәgitпјҡ
В В  В $ sudo chown -R git:git sample.git
В В  В з¬¬дә”жӯҘпјҢзҰҒз”Ёshellзҷ»еҪ•пјҡеҮәдәҺе®үе…ЁиҖғиҷ‘пјҢз¬¬дәҢжӯҘеҲӣе»әзҡ„gitз”ЁжҲ·дёҚе…Ғи®ёзҷ»еҪ•shellпјҢиҝҷеҸҜд»ҘйҖҡиҝҮзј–иҫ‘/etc/passwdж–Үд»¶е®ҢжҲҗгҖӮжүҫеҲ°зұ»дјјдёӢйқўзҡ„дёҖиЎҢgit:x:1001:1001:,,,:/home/git:/bin/bash ж”№дёәпјҡ
